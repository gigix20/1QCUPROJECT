CREATE TABLE tbl_maintenance_types (
  type_id     NUMBER,
  type_name   VARCHAR2(100) NOT NULL,
  CONSTRAINT pk_maintenance_types PRIMARY KEY (type_id)
);
CREATE SEQUENCE maint_type_seq START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;
CREATE OR REPLACE TRIGGER tbl_maint_types_bir
BEFORE INSERT ON tbl_maintenance_types
FOR EACH ROW
BEGIN
  SELECT maint_type_seq.NEXTVAL INTO :NEW.type_id FROM dual;
END;
/
COMMIT;
