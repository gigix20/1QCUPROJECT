-- BORROW/RETURN TABLE
CREATE TABLE tbl_borrows (
  borrow_id     NUMBER,
  asset_id      VARCHAR2(20)   NOT NULL,
  department_id NUMBER         NOT NULL,
  first_name    VARCHAR2(100)  NOT NULL,
  middle_name   VARCHAR2(100),
  last_name     VARCHAR2(100)  NOT NULL,
  suffix        VARCHAR2(20),
  borrow_date   VARCHAR2(20)   NOT NULL,
  due_date      VARCHAR2(20)   NOT NULL,
  return_date   VARCHAR2(20),
  purpose       VARCHAR2(255),
  status        VARCHAR2(20)   DEFAULT 'Pending',
  remarks       VARCHAR2(255),
  created_at    TIMESTAMP      DEFAULT CURRENT_TIMESTAMP,
  updated_at    TIMESTAMP      DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT pk_borrows          PRIMARY KEY (borrow_id),
  CONSTRAINT fk_borrow_asset     FOREIGN KEY (asset_id)      REFERENCES tbl_assets(asset_id),
  CONSTRAINT fk_borrow_dept      FOREIGN KEY (department_id) REFERENCES tbl_departments(department_id),
  CONSTRAINT chk_borrow_status   CHECK (status IN ('Pending', 'Borrowed', 'Returned','Overdue', 'Cancelled', 'Overdue'))
);

CREATE SEQUENCE borrow_seq START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;

CREATE OR REPLACE TRIGGER tbl_borrows_bir
BEFORE INSERT ON tbl_borrows
FOR EACH ROW
BEGIN
  SELECT borrow_seq.NEXTVAL INTO :NEW.borrow_id FROM dual;
END;
/

COMMIT;
