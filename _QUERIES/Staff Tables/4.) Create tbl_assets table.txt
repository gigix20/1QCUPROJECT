CREATE TABLE tbl_assets (
  id            NUMBER,
  asset_id      VARCHAR2(50)   NOT NULL,
  qr_code       VARCHAR2(100)  NOT NULL,
  category_id   NUMBER,
  department_id NUMBER,
  item_type_id  NUMBER,         
  description   VARCHAR2(255)  NOT NULL,
  serial_number VARCHAR2(100),
  location      VARCHAR2(150),
  status        VARCHAR2(20)   DEFAULT 'Available',
  is_certified  NUMBER(1)      DEFAULT 0,
  is_deleted    NUMBER(1)      DEFAULT 0,
  deleted_at    TIMESTAMP,
  deleted_by    VARCHAR2(100),
  created_at    TIMESTAMP      DEFAULT CURRENT_TIMESTAMP,
  updated_at    TIMESTAMP      DEFAULT CURRENT_TIMESTAMP,

  -- Constraints
  CONSTRAINT pk_assets
    PRIMARY KEY (id),
  CONSTRAINT uq_assets_asset_id
    UNIQUE (asset_id),
  CONSTRAINT uq_assets_qrcode
    UNIQUE (qr_code)
  CONSTRAINT chk_asset_certified CHECK (is_certified IN (0, 1)),
  CONSTRAINT chk_asset_deleted   CHECK (is_deleted   IN (0, 1))
);

-- Sequence for DB primary key (id)
CREATE SEQUENCE asset_id_seq
  START WITH 1
  INCREMENT BY 1
  NOCACHE
  NOCYCLE;

-- Trigger for auto id
CREATE OR REPLACE TRIGGER tbl_assets_bir
BEFORE INSERT ON tbl_assets FOR EACH ROW
BEGIN
  SELECT asset_id_seq.NEXTVAL INTO :NEW.id FROM dual;
END;
/

-- Sequence for asset_id business key (last 4 digits)
CREATE SEQUENCE asset_seq
  START WITH 1
  INCREMENT BY 1
  NOCACHE
  NOCYCLE;
