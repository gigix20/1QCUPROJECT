-- For Oracle 11g XE
CREATE TABLE password_resets (
    id NUMBER PRIMARY KEY,
    email VARCHAR2(255) NOT NULL,
    otp_hash VARCHAR2(255) NOT NULL,
    expires_at DATE NOT NULL,
    used NUMBER(1) DEFAULT 0
);

-- Sequence
CREATE SEQUENCE password_resets_seq
START WITH 1
INCREMENT BY 1
NOCACHE;

-- Trigger
CREATE OR REPLACE TRIGGER password_resets_trigger
BEFORE INSERT ON password_resets
FOR EACH ROW
BEGIN
    SELECT password_resets_seq.NEXTVAL
    INTO :NEW.id
    FROM dual;
END;
/
