CREATE TABLE tbl_item_types (
  item_type_id   NUMBER,
  item_type_name VARCHAR2(100) NOT NULL,
  item_type_code VARCHAR2(10)  NOT NULL,
  CONSTRAINT pk_item_types PRIMARY KEY (item_type_id)
);
CREATE SEQUENCE item_type_seq START WITH 1 INCREMENT BY 1;
CREATE OR REPLACE TRIGGER tbl_item_type_bir
BEFORE INSERT ON tbl_item_types FOR EACH ROW
BEGIN SELECT item_type_seq.NEXTVAL INTO :NEW.item_type_id FROM dual; END;
/
